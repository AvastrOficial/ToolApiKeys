function checkMobile(isMobile) {
    var alertMessage = document.querySelector('.popup .alert-message');
    if (isMobile) {
        alertMessage.textContent = "Espera que cargue el estilo móvil...";
        var progressBar = document.querySelector('.popup .progress-bar-inner');
        var width = 0;
        var interval = setInterval(function() {
            if (width >= 100) {
                clearInterval(interval);
                window.location.href = "https://fileboxbsz.foroactivo.com/h2-vista-movile"; // Redirigir a otra página
                return;
            }
            width += 4; // Incremento de la barra de progreso
            progressBar.style.width = width + '%';
        }, 1200); // Intervalo de actualización (en milisegundos)
    } else {
        document.querySelector('.popup').style.display = 'none';
    }
}
